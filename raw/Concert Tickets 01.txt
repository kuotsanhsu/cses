#include <iostream>
#include <set>

int main() {
    std::cin.tie(nullptr)->sync_with_stdio(false);

    std::size_t n, m;
    std::cin >> n >> m;
    std::multiset<int> hs;
    for (std::size_t i = 0; i != n; ++i) {
        int h;
        std::cin >> h;
        hs.insert(h);
    }
    for (std::size_t i = 0; i != m; ++i) {
        int t;
        std::cin >> t;
        auto H = hs.upper_bound(t);
        if (H == hs.begin()) {
            std::cout << "-1\n";
        } else {
            std::cout << *--H << "\n";
            hs.erase(H);
        }
    }
}
