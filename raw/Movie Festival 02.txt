#include <iostream>
#include <vector>
#include <algorithm>

int main() {
    std::cin.tie(nullptr)->sync_with_stdio(false);

    std::size_t n;
    std::cin >> n;
    std::vector<std::pair<int, int>> movies;
    while (n--) {
        int a, b;
        std::cin >> a >> b;
        movies.emplace_back(b, a);
    }
    std::ranges::sort(movies);
    auto count = 0;
    for (auto time = 0; const auto [b, a] : movies) {
        if (a >= time) {
            time = b;
            ++count;
        }
    }
    std::cout << count << "\n";
}
