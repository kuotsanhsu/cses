#include <iostream>
#include <map>
#include <algorithm>
 
int main() {
    std::cin.tie(nullptr)->sync_with_stdio(false);
 
    int n;
    std::cin >> n;
    int length = 0;
    std::map<int, int> mp;
    for (int i = 0, j = 1, k; std::cin >> k; ++j) {
        i = std::max(i, std::exchange(mp[k], j));
        length = std::max(length, j - i);
    }
    std::cout << length << "\n";
}
