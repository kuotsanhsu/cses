#include <iostream>
#include <iterator>
#include <array>

int main() {
    std::cin.tie(nullptr)->sync_with_stdio(false);
    const auto n = *std::istream_iterator<int>(std::cin);
    static std::array<int, 10> steps;
    for (int i = 0;;) {
        for (auto &s : steps) {
            constexpr int infinity = 1e6;
            s = infinity;
            for (int j = i + 1; j; j /= steps.size()) {
                s = std::min(s, steps[(i - j + steps.size()) % steps.size()]);
            }
            ++s;
            if (++i == n) {
                std::cout << s << "\n";
                return 0;
            }
        }
    }
}
